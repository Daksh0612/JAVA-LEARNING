Q3. Level-order traversal (Medium)

Print nodes level by level using a queue.

Waiting for your codeâ€¦
Solution â€” Level-order 
import java.util.*;
class Node{ int data; Node left,right; Node(int d){data=d; left=right=null;} }
public class Main{
  Node root;
  void levelOrder(Node root){
    if(root==null) return;
    Queue q=new LinkedList<>();
    q.add(root);
    while(!q.isEmpty()){
      Node temp=q.remove();
      System.out.print(temp.data+" ");
      if(temp.left!=null) q.add(temp.left);
      if(temp.right!=null) q.add(temp.right);
    }
  }
  public static void main(String[] args){
    Main t=new Main();
    t.root=new Node(10); t.root.left=new Node(5); t.root.right=new Node(15);
    t.levelOrder(t.root);
  }
}
