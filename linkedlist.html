class Node{ int data; Node next; Node(int d){data=d; next=null;} }
public class Main{
  Node merge(Node a, Node b){
    Node dummy=new Node(0), tail=dummy;
    while(a!=null && b!=null){
      if(a.data<=b.data){ tail.next=a; a=a.next; }
      else{ tail.next=b; b=b.next; }
      tail=tail.next;
    }
    tail.next=(a!=null)?a:b;
    return dummy.next;
  }
  void printList(Node h){ Node t=h; while(t!=null){ System.out.print(t.data+" "); t=t.next;} }

    Main ll=new Main();

    Node merged=ll.merge(l1,l2); ll.printList(merged);

class Node{ int data; Node next; Node(int d){data=d; next=null;} }
public class Main{
  boolean hasCycle(Node head){
    Node slow=head, fast=head;
    while(fast!=null && fast.next!=null){
      slow=slow.next; fast=fast.next.next;
      if(slow==fast) return true;
    }
    return false;
  }
  public static void main(String[] args){

class DNode{ int data; DNode next, prev; DNode(int d){data=d; next=prev=null;} }
public class Main{
  DNode head=null;
  void insertAtHead(int val){
    DNode n=new DNode(val);
    if(head!=null){ n.next=head; head.prev=n; }
    head=n;
  }
  void insertAtEnd(int val){
    DNode n=new DNode(val);
    if(head==null){ head=n; return; }
    DNode temp=head; while(temp.next!=null) temp=temp.next;
    temp.next=n; n.prev=temp;
  }
  void deleteHead(){ if(head!=null) head=head.next; if(head!=null) head.prev=null; }
  void printForward(){ DNode t=head; while(t!=null){ System.out.print(t.data+" "); t=t.next; } }
  public static void main(String[] args){
    Main dll=new Main();
    dll.insertAtHead(10); dll.insertAtEnd(20); dll.insertAtHead(5); dll.printForward();
  }
}
